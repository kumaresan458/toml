Java Program: MxTextBoxTest

This Java program tests the functionality of creating a text box (input field) in the context of mobile response protocols.

Program Code

import junit.framework.TestCase;

import com.mobeix.im.hub.OutputMessage;
import com.mobeix.protocol.enums.AccessibilityHeader;
import com.mobeix.protocol.enums.Cacheable;
import com.mobeix.protocol.enums.KeyScope;
import com.mobeix.protocol.enums.Mode;

public class MxTextBoxTest extends TestCase {
    /*
     * Component ID will be dynamic and passed as a parameter, retrieved from screen XML CI property.
     * masterKey is optional and can be null.
     * 
     * - If Component is created with a Master Key:
     *   - Key Scope: KeyScope.APPLICATION
     *   - Key Mode: KeyMode.NORMAL
     * 
     * - If Component is created without a Master Key:
     *   - Key Scope: KeyScope.SCREEN
     *   - Key Mode: KeyMode.AUTO
     */
    public static MxTextBox getMxTextBox(String componentId, String styleId, String textData, String masterKey) {
        MxTextBox textBox = new MxTextBox(null, textData);
        textBox.setComponentID(componentId);
        textBox.setCvKeyScope(KeyScope.SCREEN);
        textBox.setKeyMode(Mode.AUTO);
        textBox.setCvtextData(textData);
        textBox.setCvdynamicStyleId(styleId);
        textBox.setMasterKey(masterKey);
        return textBox;
    }

    public static void buildResponse(MobeixMapperOutput output) {
        MxTextBox textBox = getMxTextBox("c1", "123456", "test", null);
        output.addComponent(textBox);
    }
}
